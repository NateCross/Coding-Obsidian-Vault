# Chapter 3: Functions
## Defining a Function
```js
const square = function(x) {
    return x * x;
};

console.log(square(12));
// -> 144
```
```js
const makeNoise = function() {
    console.log("Pling!");
};

makeNoise();
// → Pling!

const power = function(base, exponent) {
    let result = 1;
    for (let count = 0; count < exponent; count++) {
        result *= base;
    }
    return result;
};

console.log(power(2, 10));
// → 1024

```
